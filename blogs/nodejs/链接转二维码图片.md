## 1.简介

把链接转换成二维码图片

## 2.安装依赖

```javascript
yarn add qr-image -S
```

## 3.实现代码

```javascript
var fs = require('fs')
var qr_image = require('qr-image');
var temp_qrcode = qr_image.image(url, {
    //设置容错率,L(低), M（中，默认）, Q（高）, H（最高）.
    ec_level: 'Q'
});
//path='images/qqwe21232.png'
temp_qrcode.pipe(fs.createWriteStream(path).on('finish', function() {
     console.log("写入成功")
 }));
```
